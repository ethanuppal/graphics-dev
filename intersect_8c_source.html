16bool intersect_aabb(struct vector src, struct vector dir,
17 const struct aabb* aabb) {
18 // Ripped from https://gamedev.stackexchange.com/a/18459
19
20 scalar_t dirfrac_x = 1.0 / dir.x;
21 scalar_t dirfrac_y = 1.0 / dir.y;
22 scalar_t dirfrac_z = 1.0 / dir.z;
23
24 scalar_t t1 = (aabb->min.x - src.x) * dirfrac_x;
25 scalar_t t2 = (aabb->max.x - src.x) * dirfrac_x;
26 scalar_t t3 = (aabb->min.y - src.y) * dirfrac_y;
27 scalar_t t4 = (aabb->max.y - src.y) * dirfrac_y;
28 scalar_t t5 = (aabb->min.z - src.z) * dirfrac_z;
29 scalar_t t6 = (aabb->max.z - src.z) * dirfrac_z;
30
31 float tmin = max(max(min(t1, t2), min(t3, t4)), min(t5, t6));
32 float tmax = min(min(max(t1, t2), max(t3, t4)), max(t5, t6));
33
34 if (tmax < 0) {
35 return false;
36 }
37
38 // if tmin > tmax, ray doesn't intersect AABB
39 if (tmin > tmax) {
40 return false;
41 }
42
43 return true;
44}

46bool intersect_face(struct vector src, struct vector dir,
47 const struct mesh* mesh, size_t face_i, scalar_t* dist,
48 struct vector* pos) {
49 // https://en.wikipedia.org/wiki/Möller–Trumbore_intersection_algorithm
50 // [dir edge1 edge2](t u v) = src - face_a
51
52 scalar_t epsilon = SCALAR_EPSILON;
53
54 struct face face = mesh_get_face(mesh, face_i);
55 struct vector face_a = mesh_get_vertex(mesh, face.v1);
56 struct vector face_b = mesh_get_vertex(mesh, face.v2);
57 struct vector face_c = mesh_get_vertex(mesh, face.v3);
58
59 struct vector edge1 = v_sub(face_b, face_a);
60 struct vector edge2 = v_sub(face_c, face_a);
61 struct vector ray_cross_e2 = v_cross(dir, edge2);
62 scalar_t det = v_dot(edge1, ray_cross_e2);
63
64 // ray is parallel : dir || edge1 or dir || edge2 so det = 0
65 if (det > -epsilon && det < epsilon) return false;
66
67 // u = det [dir (src - face_a) edge2] * invde
68 scalar_t inv_det = 1.0 / det;
69 struct vector s = v_sub(src, face_a);
70 scalar_t u = inv_det * v_dot(s, ray_cross_e2);
71
72 if (u < 0 || u > 1) return false;
73
74 struct vector s_cross_e1 = v_cross(s, edge1);
75 scalar_t v = inv_det * v_dot(dir, s_cross_e1);
76
77 if (v < 0 || u + v > 1) return false;
78
79 *dist = inv_det * v_dot(edge2, s_cross_e1);
80
81 if (*dist > epsilon) {
82 *pos = v_add(src, v_scale(dir, *dist));
83 return true;
84 } else {
85 return false;
86 }
87}
