10bool raycast(const struct mesh* scene[], size_t mesh_count, struct vector src,
11 struct vector dir, struct ray_intersection* intersect) {
12 // globally closest intersection (if one exists)
13 scalar_t closest_dist = INFINITY;
14 struct vector closest_pos;
15 const struct mesh* closest_mesh;
16 size_t closest_face_i = 0;
17
18 for (size_t k = 0; k < mesh_count; k++) {
19 const struct mesh* mesh = scene[k];
20 if (intersect_aabb(src, dir, mesh_get_aabb(mesh))) {
21 for (size_t i = 0; i < mesh_n_faces(mesh); i++) {
22 scalar_t dist;
23 struct vector pos;
24 if (intersect_face(src, dir, mesh, i, &dist, &pos)) {
25 if (dist < closest_dist) {
26 closest_dist = dist;
27 closest_pos = pos;
28 closest_mesh = mesh;
29 closest_face_i = i;
30 }
31 }
32 }
33 }
34 }
35
36 intersect->dist = closest_dist;
37 intersect->pos = closest_pos;
38 intersect->mesh = closest_mesh;
39 intersect->face_i = closest_face_i;
40
41 return closest_dist < INFINITY;
42}

44void render(const struct frame* frame, const struct scene* scene,
45 const struct camera* camera) {
46 double x_scale = camera->vb_width / (double)frame->width;
47 double y_scale = camera->vb_height / (double)frame->height;
48
49 struct vector dir = camera->focus;
50 v_normalize(&dir); // no-op
51
52 // gram schmidt to always point higher in y
53 // TODO: is allow for user defined up vector
54 struct vector h_hat = camera->focus;
55 h_hat.y += 1;
56 h_hat = v_sub(h_hat, v_scale(dir, v_dot(h_hat, dir) / v_dot(dir, dir)));
57 v_normalize(&h_hat);
58
59 struct vector w_hat = v_cross(h_hat, dir);
60 v_normalize(&w_hat); // no-op
61
62 for (size_t j = 0; j < frame->height; j++) {
63 for (size_t i = 0; i < frame->width; i++) {
64 struct vector src = v_add(
65 v_scale(w_hat, (double)i * x_scale - camera->vb_width / 2),
66 v_scale(h_hat, camera->vb_height / 2 - (double)j * y_scale));
67 src = v_add(src, camera->pos);
68
69 // you'd update dir if this was perspective and also edit it above.
70
71 frame->buffer[j * frame->width + i] = 0x0;
72
73 struct ray_intersection intersect;
74 if (raycast(scene->meshes, scene->mesh_count, src, dir,
75 &intersect)) {
76 // https://ocw.mit.edu/courses/6-837-computer-graphics-fall-2012/b86de2fe1a1084e56148f293da4a4dfd_MIT6_837F12_Lec15.pdf
77 frame->buffer[j * frame->width + i] =
78 mesh_get_face(intersect.mesh, intersect.face_i).color;
79 }
80 }
81 }
82}
