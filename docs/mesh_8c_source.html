20struct mesh* mesh_load(FILE* file) {
21 size_t n_vertices;
22 size_t n_faces;
23 if (fscanf(file, "%zu %zu\n", &n_vertices, &n_faces) < 2) {
24 mesh_load_error_string =
25 "Invalid header: expected <n_vertices> <n_faces>";
26 return NULL;
27 }
28 struct mesh* mesh = malloc(sizeof(*mesh));
29 if (!mesh) {
30 exit(1);
31 }
32
33 mesh->n_vertices = n_vertices;
34 mesh->n_faces = n_faces;
35 mesh->aabb.min.x = INFINITY;
36 mesh->aabb.min.y = INFINITY;
37 mesh->aabb.min.z = INFINITY;
38 mesh->aabb.max.x = -INFINITY;
39 mesh->aabb.max.y = -INFINITY;
40 mesh->aabb.max.z = -INFINITY;
41
42 mesh->vertices = malloc(sizeof(*mesh->vertices) * n_vertices);
43 if (!mesh->vertices) {
44 exit(1);
45 }
46
47 mesh->faces = malloc(sizeof(*mesh->faces) * n_faces);
48 if (!mesh->faces) {
49 exit(1);
50 }
51
52 for (size_t i = 0; i < n_vertices; i++) {
53 struct vector vertex;
54 if (fscanf(file, "%lf %lf %lf\n", &vertex.x, &vertex.y, &vertex.z)
55 < 3) {
56 mesh_load_error_string = "Invalid vertex: expected <x> <y> <z>";
57 return NULL;
58 }
59 mesh->vertices[i] = vertex;
60
61 if (vertex.x < mesh->aabb.min.x) {
62 mesh->aabb.min.x = vertex.x;
63 } else if (vertex.x > mesh->aabb.max.x) {
64 mesh->aabb.max.x = vertex.x;
65 }
66 if (vertex.y < mesh->aabb.min.y) {
67 mesh->aabb.min.y = vertex.y;
68 } else if (vertex.y > mesh->aabb.max.y) {
69 mesh->aabb.max.y = vertex.y;
70 }
71 if (vertex.z < mesh->aabb.min.z) {
72 mesh->aabb.min.z = vertex.z;
73 } else if (vertex.z > mesh->aabb.max.z) {
74 mesh->aabb.max.z = vertex.z;
75 }
76 }
77
78 for (size_t i = 0; i < n_faces; i++) {
79 struct face face;
80 if (fscanf(file, "%zu %zu %zu %x\n", &face.v1, &face.v2, &face.v3,
81 &face.color)
82 < 3) {
83 mesh_load_error_string =
84 "Invalid face: expected <v1> <v2> <v3> <col>";
85 return NULL;
86 }
87 mesh->faces[i] = face;
88 }
89
90 mesh_load_error_string = "No error";
91 return mesh;
92}
